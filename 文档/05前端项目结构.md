# 网络云盘系统 - 前端项目结构

## 项目概述

基于 React 18 + TypeScript + Ant Design 构建的现代化云盘前端应用，支持文件管理、实时协作、语音通话等功能。

## 技术栈

- **框架**: React 18.3.1 + TypeScript 5.6.3
- **UI库**: Ant Design 5.0+
- **状态管理**: Redux Toolkit + RTK Query
- **路由**: React Router 6.0+
- **构建工具**: Vite 6.0.1
- **样式**: Less + CSS Modules
- **实时通信**: 原生WebSocket + 自定义协议
- **音视频**: WebRTC + MediaRecorder API
- **文件处理**: spark-md5 + file-saver
- **图表**: ECharts + @ant-design/charts
- **国际化**: react-i18next
- **测试**: Jest + React Testing Library

function parsePermissions(permissions: string[]) {
  const parsedPermissions = { ...initialAuthState.permissions };
  permissions.forEach(permission => {
    const [module, action] = permission.split(':');
    if (parsedPermissions[module]) {
      parsedPermissions[module][action] = true;
    }
  });

// utils/permission.ts
export class PermissionUtils {
  // 检查单个权限
  static checkPermission(permissions: AuthState['permissions'], permission: string): boolean {
    const [module, action] = permission.split(':');
    return permissions[module]?.[action] || false;
  }
  
  // 检查多个权限（AND逻辑）
  static checkAllPermissions(permissions: AuthState['permissions'], requiredPermissions: string[]): boolean {
    return requiredPermissions.every(permission => 
      this.checkPermission(permissions, permission)
    );
  }
  
  // 检查多个权限（OR逻辑）
  static checkAnyPermission(permissions: AuthState['permissions'], requiredPermissions: string[]): boolean {
    return requiredPermissions.some(permission => 
      this.checkPermission(permissions, permission)
    );
  }
  
  // 动态渲染菜单项
  static filterMenuByPermissions(menuItems: MenuItem[], permissions: AuthState['permissions']): MenuItem[] {
    return menuItems.filter(item => {
      if (item.permission) {
        return this.checkPermission(permissions, item.permission);
      }
      if (item.children) {
        item.children = this.filterMenuByPermissions(item.children, permissions);
        return item.children.length > 0;
      }
      return true;
    });
  }
}

export const syncSlice = createSlice({
  name: 'sync',
  initialState,
  reducers: {
    startSync: (state) => {
      state.status = 'syncing';
      state.error = null;
    },
    updateProgress: (state, action) => {
      state.progress = { ...state.progress, ...action.payload };
      state.progress.percentage = Math.round((state.progress.completed / state.progress.total) * 100);
    },
    addConflict: (state, action) => {
      state.conflicts.push(action.payload);
    },
    resolveConflict: (state, action) => {
      const { conflictId, resolution } = action.payload;
      const conflict = state.conflicts.find(c => c.id === conflictId);
      if (conflict) {
        conflict.resolution = resolution;
      }
    },
    completedSync: (state) => {
      state.status = 'completed';
      state.lastSyncTime = new Date().toISOString();
      state.progress = { total: 0, completed: 0, failed: 0, percentage: 0 };
      state.conflicts = state.conflicts.filter(c => !c.resolution);
    },
    syncError: (state, action) => {
      state.status = 'error';
      state.error = action.payload;
    }
  }

      dispatch(syncSlice.actions.startSync());

      if (conflicts.length > 0) {

          dispatch(syncSlice.actions.addConflict(conflict));
        });
        return; // 等待用户解决冲突

      // 4. 增量同步
      await performIncrementalSync(localChanges, remoteChanges);
      
      dispatch(syncSlice.actions.completedSync());

      dispatch(syncSlice.actions.syncError(error.message));

      return () => clearInterval(interval);
    }
  }, [syncState.syncSettings, performSync]);
  
  return {
    performSync,
    detectLocalChanges,
    syncState
  };

      this.setupPollingWatcher(path);

      // 实现基于File System Access API的监听
    } catch (error) {
      console.warn('File System Access API not supported, falling back to polling');
      this.setupPollingWatcher(path);
    }
  }
  
  private setupPollingWatcher(path: string) {
    // 实现定时轮询的文件变更检测

    }, 5000);

3. **OpenAPI规范集成**
   ```typescript
   // tools/docs-sync.ts - 文档同步工具
   import { OpenAPIGenerator } from 'openapi-typescript-codegen';
   
   interface DocsSyncConfig {
     backendApiUrl: string;
     outputDir: string;
     modules: string[];
   }
   
   export class APISyncTool {
     async syncAPIDocs(config: DocsSyncConfig) {
       // 1. 从后端获取OpenAPI规范
       const apiSpec = await fetch(`${config.backendApiUrl}/swagger/doc.json`);
       const spec = await apiSpec.json();
       
       // 2. 按模块生成前端API文档
       for (const module of config.modules) {
         await this.generateModuleDocs(spec, module, config.outputDir);
       }
       
       // 3. 生成TypeScript类型定义
       await OpenAPIGenerator.generate({
         input: spec,
         output: `${config.outputDir}/types/`,
         clientName: 'ApiClient'
       });
       
       // 4. 验证文档一致性
       await this.validateDocsConsistency(spec, config.outputDir);
     }
     
     private async generateModuleDocs(spec: any, module: string, outputDir: string) {
       const moduleEndpoints = this.filterEndpointsByModule(spec, module);
       const markdownContent = this.generateMarkdownFromSpec(moduleEndpoints);
       
       await fs.writeFile(
         `${outputDir}/api/${module}.md`, 
         markdownContent
       );
     }
   }
   ```

   // tools/docs-validator.ts - 文档质量检查
   export class DocsValidator {
     async validateAPIConsistency() {
       const errors: string[] = [];
       
       // 1. 检查前后端API文档模块是否一致
       const frontendModules = await this.getFrontendAPIModules();
       const backendModules = await this.getBackendAPIModules();
       
       const missingModules = backendModules.filter(m => !frontendModules.includes(m));
       if (missingModules.length > 0) {
         errors.push(`前端缺少API文档模块: ${missingModules.join(', ')}`);
       }
       
       // 2. 检查API接口覆盖率
       for (const module of frontendModules) {
         const coverage = await this.checkModuleCoverage(module);
         if (coverage < 95) {
           errors.push(`${module}模块API覆盖率不足: ${coverage}%`);
         }
       }
       
       // 3. 检查文档格式规范
       await this.validateMarkdownFormat(frontendModules);
       
       return errors;
     }
   }
