# 网络云盘系统 - 系统架构设计

## 项目概述

网络云盘系统是一个基于现代化技术栈构建的企业级云存储解决方案，采用前后端分离架构，支持文件管理、实时协作、语音通话、AI智能功能等核心特性。系统设计遵循微服务架构原则，具备高可用、高性能、高扩展性的特点。

## 系统架构总览

### 架构设计原则

1. **分层架构**: 采用经典的三层架构模式，清晰分离表示层、业务逻辑层和数据访问层
2. **微服务化**: 按业务领域拆分服务，实现松耦合、高内聚的服务架构
3. **云原生**: 基于容器化部署，支持Kubernetes编排和自动扩缩容
4. **安全优先**: 全链路安全设计，包括认证授权、数据加密、访问控制等
5. **性能优化**: 多级缓存、CDN加速、数据库优化等性能提升策略
6. **可观测性**: 完整的监控、日志、链路追踪体系

```go
func (s *SearchService) SearchFiles(query string, userID uint) (*SearchResult, error) {
    var files []File
    var searchResults []SearchResult
    
    // 基础文件名搜索
    err := s.db.Select("*, MATCH(filename, file_path, tags, description) AGAINST(? IN NATURAL LANGUAGE MODE) as relevance").
        Where("owner_id = ?", userID).
        Where("MATCH(filename, file_path, tags, description) AGAINST(? IN NATURAL LANGUAGE MODE)", query, query).

        Limit(50).Find(&files).Error

        searchResults = append(searchResults, SearchResult{
            ID:       file.ID,
            Name:     file.Filename,
            Path:     file.FilePath,
            Type:     file.FileType,
            Size:     file.FileSize,
            Highlight: s.generateHighlight(file, query),
        })
    }
    
    return &SearchResults{
        Total:   len(searchResults),
        Results: searchResults,
    }, nil
}
```

```typescript
class WebSocketManager {
  private ws: WebSocket;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private heartbeatInterval: NodeJS.Timer;

  connect(token: string): Promise<void> {
    return new Promise((resolve, reject) => {
      this.ws = new WebSocket(`ws://localhost:8080/ws?token=${token}`);
      
      this.ws.onopen = () => {
        console.log('WebSocket连接成功');
        this.reconnectAttempts = 0;
        this.startHeartbeat();
        resolve();
      };
      
      this.ws.onmessage = (event) => {
        const message = JSON.parse(event.data);
        this.handleMessage(message);
      };
      
      this.ws.onclose = () => {
        this.stopHeartbeat();
        if (this.reconnectAttempts < this.maxReconnectAttempts) {
          setTimeout(() => this.reconnect(token), 1000 * Math.pow(2, this.reconnectAttempts));
          this.reconnectAttempts++;
        }
      };
    });
  }

  // 发送文件同步消息
  sendFileSync(data: FileSyncMessage): void {
    this.send({
      type: 'file_sync',
      timestamp: Date.now(),
      data: data
    });
  }

  // 发送即时消息
  sendChatMessage(data: ChatMessage): void {
    this.send({
      type: 'chat_message',
      timestamp: Date.now(),
      data: data
    });
  }

  private handleMessage(message: WebSocketMessage): void {
    switch (message.type) {
      case 'file_sync':
        this.handleFileSync(message.data);
        break;
      case 'chat_message':
        this.handleChatMessage(message.data);
        break;
      case 'system_notification':
        this.handleSystemNotification(message.data);
        break;
      case 'voice_call':
        this.handleVoiceCall(message.data);
        break;
      case 'collaborative_edit':
        this.handleCollaborativeEdit(message.data);
        break;
    }
  }
}
```

```go
// Redis消息队列服务
type MessageQueueService struct {
    client *redis.Client
}

// 发送任务到队列
func (mq *MessageQueueService) SendTask(stream string, task map[string]interface{}) error {
    return mq.client.XAdd(context.Background(), &redis.XAddArgs{
        Stream:       stream,
        MaxLen:       10000, // 最大长度
        MaxLenApprox: 10000, // 近似最大长度
        Values:       task,
    }).Err()
}

// 消费队列消息
func (mq *MessageQueueService) ConsumeTask(stream, group, consumer string) {
    for {
        streams, err := mq.client.XReadGroup(context.Background(), &redis.XReadGroupArgs{
            Group:    group,
            Consumer: consumer,
            Streams:  []string{stream, ">"},
            Count:    1,
            Block:    0, // 阻塞等待
        }).Result()
        
        if err != nil {
            log.Printf("读取消息失败: %v", err)
            time.Sleep(time.Second)
            continue
        }
        
        for _, stream := range streams {
            for _, message := range stream.Messages {
                // 处理消息
                if err := mq.processMessage(message); err != nil {
                    log.Printf("处理消息失败: %v", err)
                    // 重试逻辑
                    continue
                }
                
                // 确认消息处理完成
                mq.client.XAck(context.Background(), stream.Stream, group, message.ID)
            }
        }
    }
}

// 发布订阅模式 - 实时通知
func (mq *MessageQueueService) PublishEvent(channel string, data interface{}) error {
    jsonData, _ := json.Marshal(data)
    return mq.client.Publish(context.Background(), channel, jsonData).Err()
}

// 订阅实时事件
func (mq *MessageQueueService) SubscribeEvent(channel string, handler func([]byte)) {
    pubsub := mq.client.Subscribe(context.Background(), channel)
    defer pubsub.Close()
    
    for msg := range pubsub.Channel() {
        handler([]byte(msg.Payload))
    }
}
```

```typescript
class SmartChunkUploader {
  private getOptimalChunkSize(fileSize: number, networkSpeed: number): number {
    // 根据文件大小和网络速度动态调整分片大小
    if (fileSize < 10 * 1024 * 1024) { // 小于10MB
      return Math.min(fileSize, 2 * 1024 * 1024); // 最大2MB
    } else if (fileSize < 100 * 1024 * 1024) { // 小于100MB
      return networkSpeed > 1024 * 1024 ? 5 * 1024 * 1024 : 2 * 1024 * 1024;
    } else { // 大于100MB
      return networkSpeed > 5 * 1024 * 1024 ? 10 * 1024 * 1024 : 5 * 1024 * 1024;
    }
  }
  
  private async uploadChunkWithRetry(chunk: Blob, chunkIndex: number, maxRetries = 3): Promise<void> {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
      try {
        await this.uploadChunk(chunk, chunkIndex);
        return;
      } catch (error) {
        if (attempt === maxRetries) {
          throw error;
        }
        // 指数退避重试
        await this.delay(Math.pow(2, attempt) * 1000);
      }
    }
  }
}
```

```go
        httpRequestDuration.WithLabelValues(c.Request.Method, c.FullPath()).Observe(duration)
    })
}
```
